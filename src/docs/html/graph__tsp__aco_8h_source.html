70 TsmResult Run() {
71 TsmResult best_result;
72 best_result.vertices = {};
73 best_result.distance = std::numeric_limits<double>::max();
74
75 for (int iter = 0; iter < params_.num_iterations; ++iter) {
76 std::vector<std::vector<int>> ant_tours(params_.num_ants);
77 std::vector<double> ant_tour_lengths(params_.num_ants);
78
79 for (int k = 0; k < params_.num_ants; ++k) {
80 ant_tours[k] = ConstructTourForAnt();
81 if (ant_tours[k].empty()) continue; // Skip if ant got stuck
82
83 ant_tour_lengths[k] = CalculateTourLength(ant_tours[k]);
84
85 if (ant_tour_lengths[k] < best_result.distance) {
86 best_result.distance = ant_tour_lengths[k];
87 best_result.vertices = ant_tours[k];
88 // std::cout << "Iter " << iter << ", Ant " << k << ": New best tour
89 // length: " << best_result.distance << std::endl;
90 }
91 }
92
93 UpdatePheromones(ant_tours, ant_tour_lengths);
94 }
95
96 // Ensure the result tour is valid (not empty)
97 if (best_result.vertices.empty() && num_cities_ > 0) {
98 throw std::runtime_error("ACO could not find a valid tour.");
99 }
100
101 // Add the starting node to the end to represent the full cycle for TsmResult
102 if (!best_result.vertices.empty()) {
103 best_result.vertices.push_back(best_result.vertices.front());
104 }
105

162 std::vector<int> ConstructTourForAnt() {
163 std::vector<int> tour;
164 tour.reserve(num_cities_);
165 std::vector<bool> visited(num_cities_, false);
166
167 // Start at a random city
168 std::uniform_int_distribution<> start_dist(0, num_cities_ - 1);
169 int current_city = start_dist(random_generator_);
170 tour.push_back(current_city);
171 visited[current_city] = true;
172
173 while (tour.size() < num_cities_) {
174 int next_city = SelectNextCity(current_city, visited);
175 if (next_city == -1) {
176 // Ant got stuck - should not happen in a complete graph TSP
177 return {}; // Return empty tour to indicate failure
178 }
179 tour.push_back(next_city);
180 visited[next_city] = true;
181 current_city = next_city;
182 }
183
184 return tour;
185 }

196 void UpdatePheromones(const std::vector<std::vector<int>>& ant_tours,
197 const std::vector<double>& ant_tour_lengths) {
198 // 1. Evaporation
199 for (int i = 0; i < num_cities_; ++i) {
200 for (int j = i + 1; j < num_cities_;
201 ++j) { // Iterate only upper triangle for symmetric TSP
202 pheromones_[i][j] *= (1.0 - params_.evaporation_rate);
203 pheromones_[j][i] = pheromones_[i][j];
204 }
205 }
206
207 // 2. Deposition
208 for (size_t k = 0; k < ant_tours.size(); ++k) {
209 const auto& tour = ant_tours[k];
210 if (tour.empty()) continue; // Skip invalid tours
211
212 double tour_length = ant_tour_lengths[k];
213 if (tour_length <= 0 || std::isinf(tour_length) ||
214 std::isnan(tour_length)) {
215 // std::cerr << "Warning: Skipping ant with invalid tour length: " <<
216 // tour_length << std::endl;
217 continue; // Skip invalid tour lengths
218 }
219
220 double pheromone_deposit = params_.Q / tour_length;
221
222 for (size_t i = 0; i < tour.size() - 1; ++i) {
223 int city1 = tour[i];
224 int city2 = tour[i + 1];
225 pheromones_[city1][city2] += pheromone_deposit;
226 pheromones_[city2][city1] = pheromones_[city1][city2];
227 }
228 // Deposit on the return edge
229 int last_city = tour.back();
230 int first_city = tour.front();
231 pheromones_[last_city][first_city] += pheromone_deposit;
232 pheromones_[first_city][last_city] = pheromones_[last_city][first_city];
233 }
234 }

246 double CalculateTourLength(const std::vector<int>& tour) const {
247 double length = 0.0;
248 if (tour.size() < 2)
249 return std::numeric_limits<double>::infinity(); // Not a valid tour
250
251 for (size_t i = 0; i < tour.size() - 1; ++i) {
252 double dist = graph_(tour[i], tour[i + 1]);
253 if (dist <= 0 || std::isinf(dist)) {
254 return std::numeric_limits<double>::infinity(); // Invalid edge in tour
255 }
256 length += dist;
257 }
258 // Add distance from last city back to the first
259 double return_dist = graph_(tour.back(), tour.front());
260 if (return_dist <= 0 || std::isinf(return_dist)) {
261 return std::numeric_limits<double>::infinity(); // Invalid return edge
262 }
263 length += return_dist;
264 return length;
265 }

278 int SelectNextCity(int current_city, const std::vector<bool>& visited) {
279 double total_attractiveness = 0.0;
280 std::vector<double> probabilities;
281 std::vector<int> allowed_cities;
282 probabilities.reserve(num_cities_);
283 allowed_cities.reserve(num_cities_);
284
285 for (int next_city = 0; next_city < num_cities_; ++next_city) {
286 if (!visited[next_city]) {
287 double pheromone_level = pheromones_[current_city][next_city];
288 double heuristic_level = heuristic_info_[current_city][next_city];
289
290 // Consider only reachable cities with some attractiveness
291 if (pheromone_level > 1e-9 ||
292 heuristic_level > 1e-9) { // Use a small threshold
293 double attractiveness = std::pow(pheromone_level, params_.alpha) *
294 std::pow(heuristic_level, params_.beta);
295 if (attractiveness < 1e-9) {
296 attractiveness = 1e-9; // Prevent zero probability if possible
297 }
298 probabilities.push_back(attractiveness);
299 allowed_cities.push_back(next_city);
300 total_attractiveness += attractiveness;
301 }
302 }
303 }
304
305 if (allowed_cities.empty()) {
306 return -1; // No valid next city found
307 }
308
309 // Roulette wheel selection
310 std::uniform_real_distribution<> dist(0.0, total_attractiveness);
311 double roulette_spin = dist(random_generator_);
312 double cumulative_prob = 0.0;
313
314 for (size_t i = 0; i < allowed_cities.size(); ++i) {
315 cumulative_prob += probabilities[i];
316 if (roulette_spin <= cumulative_prob) {
317 return allowed_cities[i];
318 }
319 }
320
321 // In case of floating point issues
322 return allowed_cities.back();
323 }
